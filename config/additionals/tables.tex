% Change spaces of tables
\setlength{\tabcolsep}{10pt}
\renewcommand{\arraystretch}{1.5}


%%%%%%%%%% Create new save command %%%%%%%%%%

\makeatletter

% \savetable[<OPTIONAL>]{<ID>}{<adjust_settings>}{<column_setting>}{<description>}{<content>}

% OPTIONAL: width of table lines e.g. 2pt
% adjust_settings: any settings for \begin{adjustbox} e.g. center
% column_setting: any column settings for \begin{tabular} e.g. |c|c|c|c|c|c|
% description: caption to show below the table
% content: the content of the table
\newcommand\savetable[6][]{%
    \@namedef{tab@#2}{{
        % Change width of table lines, if set
        \ifthenelse{\isempty{#1}}
            {}% True
            {\setlength\arrayrulewidth{#1}}%False

        \begin{adjustbox}{#3, captionbelow={#5\label{tab:#2}}, nofloat=table}
            \begin{tabular}[c]{#4}
                \hline
                #6
            \end{tabular}
        \end{adjustbox}
    }}
}

% \usetable{<ID>}
\newcommand\usetable[1]{%
    \@nameuse{tab@#1}
}
